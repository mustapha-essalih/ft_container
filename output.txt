#include <iostream>
#include <fstream>
#include <vector>
#include <iterator>

int main()
{
    std::vector<int> vec{ 1, 2, 3, 4, 5 };
    std::ofstream ofs("output.txt");

    std::ostream_iterator<int> output_iterator(ofs, "\n");

    std::copy(vec.begin(), vec.end(), output_iterator);
    ofs.close();

    return 0;
}
